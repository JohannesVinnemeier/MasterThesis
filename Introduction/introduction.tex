\section{Introduction/Background}

\begin{definition}
    A closed oriented $n$-manifold $M$ is called enlargeable if there exists a Riemannian metric on $M$, such that for all $\epsilon >0$ there exists a finite covering $\hat{M} \to M$ together with a non-zero degree $\epsilon$-contracting map $\hat{M} \to S^n$.
\end{definition}

Note that this notion does not depend on the Riemannian metric on $M$. Our definition of enlargeabilty is adopted from \cite{cecchini_schick} in contrast to the orginial definition by Gromov and Lawson in \cite{gromov_lawson}, where the covering $\hat{M}$ was additionally required to be spin. 

The following Lemma provides two easy ways to construct new enlargeable manifolds out of given ones.

\begin{lemma}
    \label{lem:constructing_enlargeable_mnf}
    \begin{enumerate}[label=\roman*)]
        \item The product of enlargeable manifolds is enlargeable.
        \item A closed oriented manifold $M$ admitting a non-zero degree map onto an enlargeable manifold is enlargeable.
    \end{enumerate}
\end{lemma}

\begin{theorem}[Gromov, Lawson and Cecchini, Schick]
    Enlargeable manifolds do not admit metrics of positive scalar curvature.
\end{theorem}
